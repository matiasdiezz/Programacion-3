class App extends Component {
  constructor (props) {
      super(props);
      this.state = {
          filterBy: ''
      };
  }
  filtrarArtistas(textoAFiltrar) {
      this.setState({
          filterBy: textoAFiltrar
      },()=>this.filtrarArtistas(textoAFiltrar)
      );
  }

  render() {
  return (
    <div>
      <div>
      <Header filterBy={(params)=>this.filtrarArtistas(params)}/>
      </div>
      <div className='contenedor'>
      <Artistas/>
      </div>
    </div>
  );
}
}

export default App;
